\documentclass{standalone}

\usepackage{tikz}
\usetikzlibrary{shapes,arrows,chains}

\begin{document}
	
% -------------------------------------------------
% Set up a new layer for the debugging marks, and make sure it is on
% top
\pgfdeclarelayer{marx}
\pgfsetlayers{main,marx}
% A macro for marking coordinates (specific to the coordinate naming
% scheme used here). Swap the following 2 definitions to deactivate
% marks.
\providecommand{\cmark}[2][]{%
  \begin{pgfonlayer}{marx}
    \node [nmark] at (c#2#1) {#2};
  \end{pgfonlayer}{marx}
  } 
\providecommand{\cmark}[2][]{\relax} 

\tikzset{
  base/.style={draw, on chain, on grid, align=center, minimum height=4ex},
  input/.style={base, rectangle, text width=8em, draw=none},
  question/.style={base, diamond, aspect=2, text width=5em},
  operator/.style={base, rounded corners, rectangle, text width=8em},
  % coord node style is used for placing corners of connecting lines
  coord/.style={coordinate, on chain, on grid, node distance=6mm and 25mm},
  % nmark node style is used for coordinate debugging marks
  nmark/.style={draw, cyan, circle, font={\sffamily\bfseries}},
  % -------------------------------------------------
  % Connector line styles for different parts of the diagram
  norm/.style={->, draw},
  free/.style={->, draw},
  cong/.style={->, draw},
  it/.style={font={\small\itshape}}
}

% -------------------------------------------------
% Start the picture
\begin{tikzpicture}[%
    >=triangle 60,              % Nice arrows; your taste may be different
    start chain=going below,    % General flow is top-to-bottom
    node distance=6mm and 60mm, % Global setup of box spacing
    every join/.style={norm},   % Default linetype for connecting boxes
    ]



	\node [input] 													                     {word image};
	\node [operator, join] 				(o_gen_ssps)							     {generate SSPs};
	\node [input, join] 											                   {set of SSPs};
	\node [operator, join] 											                 {select one SSP};
	\node [input, join]												                   {segmentation point (SP)};
	\node [input, join] 											                   {two sub-images};
	\node [question, join] 		                  (q_terminate)				     {terminate?};
	\node [operator]					                  (q_terminate_n_answer)	 {take one sub-image};
  \node [input, join, left=of q_terminate_n_answer]    (image)                  {a sub image};
	\node [input, right=of q_terminate]         (q_terminate_y_answer) 	 {a set of sub images};

	% Question node arrows
  \path (q_terminate.east) to node [near start, xshift=1em, yshift=1ex] {yes} (q_terminate_y_answer);
  \draw [*->] (q_terminate.east) -- (q_terminate_y_answer);
  \path (q_terminate.south) to node [near start, xshift=1em, yshift=-1ex] {no} (q_terminate_n_answer);
  \draw [*->] (q_terminate.south) -- (q_terminate_n_answer);

  %Helper nodes
  \node [coord, above=of image] (c1)  {}; \cmark{1}

  % Paths with angle
  \path (image.north) to node [yshift=-1em] {} (c1); 
  \draw [*->] (image.north) -- (c1) |- (o_gen_ssps);

\end{tikzpicture}
